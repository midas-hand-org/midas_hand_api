194    def wait_until_reached(
195        self,
196        positions_rad: Sequence[float],
197        tolerance_rad: float = 0.03,
198        velocity_threshold_rad_s: Optional[float] = None,
199        timeout_s: float = 5.0,
200        poll_interval_s: float = 0.02,
201        contact_current_ma: Optional[float] = None,
202    ) -> bool:
203        """Wait until active motors are near ``positions_rad``.
204
205        Returns ``True`` when all active motors are within ``tolerance_rad`` and,
206        if requested, below ``velocity_threshold_rad_s``. Returns ``False`` on
207        timeout or contact-current early exit.
208        """
209
210        target = np.asarray(positions_rad, dtype=np.float64)
211        if len(target) != len(self.motor_ids):
212            raise ValueError(f"Expected {len(self.motor_ids)} positions")
213
214        deadline = time.monotonic() + timeout_s
215        while time.monotonic() < deadline:
216            pos, vel, cur = self._read_motor_pos_vel_cur()
217            position_reached = np.all(np.abs(target - pos) <= tolerance_rad)
218            velocity_settled = (
219                True
220                if velocity_threshold_rad_s is None
221                else np.all(np.abs(vel) <= velocity_threshold_rad_s)
222            )
223            if position_reached and velocity_settled:
224                return True
225            if contact_current_ma is not None and np.any(
226                np.abs(cur) >= contact_current_ma
227            ):
228                return False
229            time.sleep(poll_interval_s)
230        return False

336    def _expand_positions(self, motor_pos: np.ndarray) -> np.ndarray:
337        """Expand 13-motor positions to 16-DOF joint space.
338
339        Passive DIP joints are inserted at ``config.passive_joint_indices``
340        and estimated from their coupled PIP motors via
341        :func:`~midas_hand_api.kinematics.pip_to_dip_position`.
342        """
343        result = np.empty(self.config.n_dof, dtype=np.float64)
344        passive_map = dict(zip(self.config.passive_joint_indices, self.config.pip_motor_indices))
345        motor_idx = 0
346        for dof_idx in range(self.config.n_dof):
347            if dof_idx in passive_map:
348                result[dof_idx] = kinematics.pip_to_dip_position(
349                    motor_pos[passive_map[dof_idx]]
350                )
351            else:
352                result[dof_idx] = motor_pos[motor_idx]
353                motor_idx += 1
354        return result

377            else:

379                motor_idx += 1
