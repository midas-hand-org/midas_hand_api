 39def _repeat(value: float, n: int) -> Tuple[float, ...]:
 40    return tuple(value for _ in range(n))

 43def _coerce_motor_values(
 44    values: Union[Sequence[float], Mapping[Union[int, str], float]],
 45    motor_ids: Sequence[int],
 46    name: str,
 47) -> Tuple[float, ...]:
 48    """Return motor-indexed values ordered to match ``motor_ids``.
 49
 50    Saved configs may store calibration fields as ``{motor_id: value}`` for
 51    robustness. Runtime math uses tuples/arrays aligned with ``motor_ids``.
 52    """
 53
 54    if isinstance(values, Mapping):
 55        ordered = []
 56        for motor_id in motor_ids:
 57            if motor_id in values:
 58                ordered.append(float(values[motor_id]))
 59            elif str(motor_id) in values:
 60                ordered.append(float(values[str(motor_id)]))
 61            else:
 62                raise ValueError(f"{name} is missing value for motor ID {motor_id}")
 63        return tuple(ordered)
 64    return tuple(float(value) for value in values)

 67def _motor_value_map(
 68    motor_ids: Sequence[int], values: Sequence[float]
 69) -> dict[int, float]:
 70    return {int(motor_id): float(value) for motor_id, value in zip(motor_ids, values)}

 73def _lookup_motor_value(
 74    values: Mapping[Union[int, str], float],
 75    motor_id: int,
 76    name: str,
 77) -> float:
 78    if motor_id in values:
 79        return float(values[motor_id])
 80    key = str(motor_id)
 81    if key in values:
 82        return float(values[key])
 83    raise ValueError(f"{name} is missing value for motor ID {motor_id}")

 86def _filtered_passive_topology(
 87    source_n_motors: int,
 88    source_passive_joint_indices: Sequence[int],
 89    source_pip_motor_indices: Sequence[int],
 90    selected_motor_indices: Sequence[int],
 91) -> tuple[tuple[int, ...], tuple[int, ...]]:
 92    selected_old_to_new = {
 93        old_index: new_index
 94        for new_index, old_index in enumerate(selected_motor_indices)
 95    }
 96    passive_by_dof = dict(zip(source_passive_joint_indices, source_pip_motor_indices))
 97
 98    passive_joint_indices = []
 99    pip_motor_indices = []
100    motor_index = 0
101    filtered_dof_index = 0
102    for dof_index in range(source_n_motors + len(source_passive_joint_indices)):
103        if dof_index in passive_by_dof:
104            pip_motor_index = passive_by_dof[dof_index]
105            if pip_motor_index in selected_old_to_new:
106                passive_joint_indices.append(filtered_dof_index)
107                pip_motor_indices.append(selected_old_to_new[pip_motor_index])
108                filtered_dof_index += 1
109        else:
110            if motor_index in selected_old_to_new:
111                filtered_dof_index += 1
112            motor_index += 1
113
114    return tuple(passive_joint_indices), tuple(pip_motor_indices)

119    """Runtime configuration for a Dynamixel-driven hand.
120
121    The defaults are intentionally conservative placeholders for a 13-motor hand.
122    Update ``motor_ids``, ``home_offsets``, ``joint_signs``, and limits after
123    calibration against the real Midas hand.

126    motor_ids: Tuple[int, ...] = DEFAULT_MOTOR_IDS

136    #   drives each corresponding passive DIP (same order).

212    def validate(self) -> None:
213        n = len(self.motor_ids)
214        motor_fields = {
215            name: getattr(self, name)
216            for name in _MOTOR_VALUE_FIELDS
217        }
218        for name, values in motor_fields.items():
219            if len(values) != n:
220                raise ValueError(f"{name} must have {n} values (one per motor), got {len(values)}")
221        if len(self.passive_joint_indices) != len(self.pip_motor_indices):
222            raise ValueError(
223                "passive_joint_indices and pip_motor_indices must have the same length"
224            )
225        for idx in self.pip_motor_indices:
226            if idx >= n:
227                raise ValueError(
228                    f"pip_motor_indices contains {idx}, but only {n} motors are configured"
229                )
230        if self.goal_current_limit > self.max_goal_current_limit:
231            raise ValueError(
232                "goal_current_limit must be <= "
233                f"{self.max_goal_current_limit} for {self.motor_model}"
234            )

254            selected_default_indices,
255        )

258            "motor_model": "XM335-T323-T",
259            "expected_model_number": ct.XM335_T323_MODEL_NUMBER,
260            "port": port,
261            "baudrate": baudrate,
262            "position_p_gain": 900,
263            "position_i_gain": 0,
264            "position_d_gain": 600,
265            "goal_current_limit": 600,
266            "max_goal_current_limit": ct.XM335_T323_MAX_CURRENT_LIMIT,
267            "operating_mode": ct.OPERATING_MODE_CURRENT_BASED_POSITION,
268            "counts_per_rev": 4096,
269            "velocity_unit_rpm": 0.229,
270            "current_unit_ma": ct.XM335_T323_CURRENT_UNIT_MA,
271            "home_offsets": _repeat(0.0, n),
272            "joint_signs": _repeat(1.0, n),
273            "joint_lower_limits": _repeat(-pi, n),
274            "joint_upper_limits": _repeat(pi, n),
275            "passive_joint_indices": passive_joint_indices,
276            "pip_motor_indices": pip_motor_indices,
277        }
278        values.update(overrides)
279        return cls(**values)

281    def subset(self, motor_ids: Sequence[int]) -> "HandConfig":
282        """Return this config narrowed to ``motor_ids`` while preserving calibration."""
283
284        requested_ids = set(int(motor_id) for motor_id in motor_ids)
285        selected_ids = tuple(
286            motor_id for motor_id in self.motor_ids if motor_id in requested_ids
287        )
288        selected_indices = [
289            index
290            for index, motor_id in enumerate(self.motor_ids)
291            if motor_id in requested_ids
292        ]
293        if len(selected_ids) != len(requested_ids):
294            missing = sorted(requested_ids - set(selected_ids))
295            raise ValueError(f"Cannot subset config for unknown motor IDs: {missing}")
296
297        def select(values):
298            return tuple(values[index] for index in selected_indices)
299
300        passive_joint_indices, pip_motor_indices = _filtered_passive_topology(
301            self.n_motors,
302            self.passive_joint_indices,
303            self.pip_motor_indices,
304            selected_indices,
305        )
306        return replace(
307            self,
308            motor_ids=selected_ids,
309            passive_joint_indices=passive_joint_indices,
310            pip_motor_indices=pip_motor_indices,
311            **{name: select(getattr(self, name)) for name in _MOTOR_VALUE_FIELDS},
312        )

314    def save(self, path: Union[str, pathlib.Path] = DEFAULT_CONFIG_PATH) -> None:
315        """Save hand calibration to YAML (default: ``~/.midas_hand/config.yaml``).
316
317        The YAML intentionally stores only hand-specific calibration values.
318        Controller gains, current limits, model constants, and baudrate live in
319        code defaults and can be overridden explicitly at runtime.
320        """
321        path = pathlib.Path(path)
322        path.parent.mkdir(parents=True, exist_ok=True)
323        data = {
324            "motor_ids": list(self.motor_ids),
325            "home_offsets": _motor_value_map(self.motor_ids, self.home_offsets),
326            "joint_signs": _motor_value_map(self.motor_ids, self.joint_signs),
327            "joint_lower_limits": _motor_value_map(
328                self.motor_ids, self.joint_lower_limits
329            ),
330            "joint_upper_limits": _motor_value_map(
331                self.motor_ids, self.joint_upper_limits
332            ),
333        }
334        with open(path, "w") as f:
335            yaml.dump(data, f, default_flow_style=False, sort_keys=False)

337    def save_merged_offsets(
338        self,
339        motor_indices: Sequence[int],
340        path: Union[str, pathlib.Path] = DEFAULT_CONFIG_PATH,
341    ) -> None:
342        """Merge selected home offsets into the saved calibration YAML.
343
344        Homing may be run on only part of the hand. This method preserves
345        existing calibration entries for other motors, updates/replaces only
346        the successfully homed motor IDs, and drops stale runtime tuning fields.
347        """
348
349        path = pathlib.Path(path)
350        if path.exists():
351            with open(path) as f:
352                data = yaml.safe_load(f) or {}
353        else:
354            data = {}
355
356        saved_motor_ids = [int(motor_id) for motor_id in data.get("motor_ids", [])]
357        existing_ids = list(saved_motor_ids)
358        for motor_id in self.motor_ids:
359            if int(motor_id) not in existing_ids:
360                existing_ids.append(int(motor_id))
361
362        merged_data = {"motor_ids": existing_ids}
363
364        for key in _MOTOR_VALUE_FIELDS:
365            values = getattr(self, key)
366            existing = data.get(key, {})
367            if isinstance(existing, list):
368                existing = {
369                    int(motor_id): float(value)
370                    for motor_id, value in zip(saved_motor_ids, existing)
371                }
372            else:
373                existing = {
374                    int(motor_id): float(value)
375                    for motor_id, value in existing.items()
376                }
377
378            if key == "home_offsets":
379                indices_to_write = motor_indices
380                for index in indices_to_write:
381                    existing[int(self.motor_ids[index])] = float(values[index])
382            else:
383                for index, motor_id in enumerate(self.motor_ids):
384                    existing.setdefault(int(motor_id), float(values[index]))
385            merged_data[key] = existing
386
387        path.parent.mkdir(parents=True, exist_ok=True)
388        with open(path, "w") as f:
389            yaml.dump(merged_data, f, default_flow_style=False, sort_keys=False)

391    @classmethod
392    def load(cls, path: Union[str, pathlib.Path] = DEFAULT_CONFIG_PATH) -> "HandConfig":
393        """Load calibration from YAML over the code defaults.
394
395        Runtime fields that may exist in older YAML files, such as gains,
396        current limits, port, baudrate, and model constants, are ignored.
397        """
398        with open(path) as f:
399            data = yaml.safe_load(f) or {}
400        motor_ids = tuple(
401            int(motor_id) for motor_id in data.get("motor_ids", DEFAULT_MOTOR_IDS)
402        )
403        config = cls.xm335_t323(motor_ids=motor_ids)
404        updates = {}
405        for key in _MOTOR_VALUE_FIELDS:
406            if key in data:
407                val = data[key]
408                if isinstance(val, dict):
409                    updates[key] = tuple(
410                        _lookup_motor_value(val, int(mid), key) for mid in motor_ids
411                    )
412                elif isinstance(val, list):
413                    updates[key] = tuple(float(value) for value in val)
414        if updates:
415            config = replace(config, **updates)
416        return config
